using namespace std;

int main()

{
    int t;
    cin>>t;
    
    while(t--)
    {
        int p,q;
        cin>>p>>q;
        
        int count=0;
        
        for(int i=1;i<=p;i++)
        {
            if(p%i==0 && q%i==0)
            {
                count++;
            }
        }
        cout<<count<<endl;
    }
}
